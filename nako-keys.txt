-K / --keyring キーリングの管理

-i / --init キーリングの初期化
sudo rm /etc/pacman.d/gnupg
sudo pacman-key --init

-p / --populate マスター鍵の検証
sudo pacman-key --populate <keyname>

-r / --recv サーバーから鍵を取得
--keyserver 鍵サーバー
sudo pacman-key --recv-keys <keyid> --keyserver <keyserver>

-f / --finger 鍵の指紋の確認
pacman-key --finger <keyid>

-s / --sign 取得した鍵に署名
sudo pacman-key --lsign-key <keyid>

-l / --list 鍵一覧の表示
pacman-key --list-keys

-a / --add 鍵ファイルを追加
sudo pacman-key --add <keypath>

-d / --delete 鍵を削除
sudo pacman-key --delete <keyid>
